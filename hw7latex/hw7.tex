\[ 0 = \frac{Q_H}{T_H} + \frac{Q_L}{T_L} - \frac{Q_m}{T_m} \]

\[ Q_m = Q_H + Q_L   \]

\[ 0 = \frac{Q_H}{T_H} + \frac{Q_L}{T_L} - \frac{Q_H + Q_L}{T_m} = \frac{Q_H}{T_H} - \frac{Q_H}{T_m} + \frac{Q_L}{T_L} + \frac{Q_L}{T_m} \]
